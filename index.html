jQuery Create Url

Basic Usage

Initiate global configurations.

$.UrlConfig([
    {
        pattern: '<type>-cars/<make>/<model>/<variant>',
        route: 'listings/index',
        defaults: {
            vehicle_type: 'car'
        }
    },
    {
        pattern: 'motorbike/<make>/<model>/<variant>',
        route: 'listings/index',
        defaults: {
            vehicle_type: 'mcycle'
        }
    },
    {
        pattern: '<make>/<model>/<variant>',
        route: 'listings/index',
    }
]);

Create urls.

var url1 = $.Url('listings/index', {
    make: 'proton',
    model: 'saga',
    variant: 'blm',
    vehicle_type: 'mcycle',
});
console.log(url1);

var url2 = $.Url('listings/index', {
    make: 'proton',
    model: 'saga',
    variant: 'blm'
});
console.log(url2);

var url3 = $.Url('book/article', {
    category: 'cooking',
    title: 'this-is-title',
    id: 12345678
});
console.log(url3);
